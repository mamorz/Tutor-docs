\section{Aufgabe: Temperatur}

\begin{frame}{Aufgabe: Temperatur}
    \begin{block}{Schreiben Sie eine Klasse \textit{Temperature}, die \dots}
        \begin{itemize}
            \item eine Temperatur enthält
            \item Möglichkeiten zur Ausgabe in bekannten Einheiten ermöglicht
            \item Konstruktoren, getter und setter je nach Notwendigkeit
            \item einem enum mit Temperaturkategorien enthält, denen Sie einem Temperaturbereich zuordnen
            \item eine geeignete Ausgabe der Temperaturkategorie
            \item eine \textit{equals()}-, \textit{hashCode()}- und eine \textit{toString()}-Methode besitzt
        \end{itemize}
    \end{block}
\end{frame}
