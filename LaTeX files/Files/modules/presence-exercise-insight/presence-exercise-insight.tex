\section{Präsenzübung}

\begin{frame}{Präsenzübung - Einsicht}
    \begin{block}{Formalitäten}
        \begin{itemize}
            \item Im Austausch gegen den Studentausweis
            \item Jeweils nur 3 Personen gleichzeitig in der ersten Reihe
            \item Keine Fotos, Kopien, Mitschriften o.ä. (Stifte \& Handy weg)
            \item Musterlösung gibt es vorne
            \item Bei Fragen bescheid geben
            \item Bitte Verzicht auf Änderung, wenn es nicht um Bestehen/Nicht Bestehen geht
        \end{itemize}
    \end{block}
\end{frame}
